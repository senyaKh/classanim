Структуры Автомобилей на C++

Задача: Создать структуры автомобиля с одинаковыми свойствами и разными цветами

Код
.cpp
#include <iostream>
using namespace std;

// Структура автомобиля
struct Car {
    string model;
    int wheels;
    string doors;
    string headlights;
    string hood;
    string color;
};

int main() {
    // Создание структур автомобилей
    Car car1 = {"Car Model 1", 4, "Standard", "LED", "Metal", "Red"};
    Car car2 = {"Car Model 2", 4, "Standard", "LED", "Metal", "Green"};
    Car car3 = {"Car Model 3", 4, "Standard", "LED", "Metal", "Blue"};

    // Вывод информации о car1
    cout << "Model: " << car1.model << ", Color: " << car1.color << "\n";

    // Вывод информации о car2
    cout << "Model: " << car2.model << ", Color: " << car2.color << "\n";

    // Вывод информации о car3
    cout << "Model: " << car3.model << ", Color: " << car3.color << "\n";

    return 0;
}
